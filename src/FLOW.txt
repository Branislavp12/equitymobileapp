Importy: z čoho súbor importuje a načo to používa

Exporty: čo konkrétne poskytuje iným súborom (napr. funkcie, komponenty, hooky)

Poznámka: čo sa v tom súbore udeje (hlavná zodpovednosť)




- file: AppNavigator.js
  imports: [WelcomeScreen.js, TradingScreen.js]
  exports: [AppNavigator component]
  purpose: "Definuje navigáciu medzi obrazovkami Welcome a Trading."

- file: WelcomeScreen.js
  imports: [persist.js]
  exports: [WelcomeScreen component]
  purpose: "Zobrazuje vstup pre počiatočný kapitál. Ukladá ho do AsyncStorage. Redirectuje do Trading."

- file: TradingScreen.js
  imports: [useTradeHistory.js, persist.js, TradeForm.js, EquityChart/index.js]
  exports: [TradingScreen component]
  purpose: "Hlavná obrazovka: zobrazuje graf, spravuje obchody, formulár, štatistiky a akcie (reset, undo)."

- file: useTradeHistory.js
  imports: [persist.js]
  exports: [useTradeHistory hook]
  purpose: "Centralizovaná logika obchodov: kapitál, obchody, výpočet equity, štatistiky, undo/reset."

- file: TradeForm.js
  imports: [@ui-kitten/components]
  exports: [TradeForm component]
  purpose: "Formulár na zadanie risk/reward, validácia vstupu, výstraha pri veľkom riske."

- file: EquityChart/index.js

  imports: []
